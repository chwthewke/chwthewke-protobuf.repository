271         final File artifactFile = artifact.getFile( );

307     private void extractFile( final File file, final String type, final File absoluteExtractPath )
308             throws MojoExecutionException {
309         try
310         {
311             final UnArchiver unarchiver = archiverManager.getUnArchiver( type );
312             unarchiver.setSourceFile( file );
313             unarchiver.setDestDirectory( absoluteExtractPath );
314             unarchiver.setFileSelectors( new FileSelector[ ] { new FileSelector( ) {
315 
316                 @Override
317                 public boolean isSelected( final FileInfo fileInfo ) throws IOException {
318                     return !fileInfo.getName( ).startsWith( "META-INF" );
319                 }
320             } } );
321 
322             //getLog( ).info( "Unarchiver: " + unarchiver );
323 
324             unarchiver.extract( );
325         }
326         catch ( final NoSuchArchiverException e )
327         {
328             throw new MojoExecutionException( String.format( "Cannot unpack file %s.", file ), e );
329         }
330         catch ( final ArchiverException e )
331         {
332             throw new MojoExecutionException( String.format( "Failed to unpack file %s.", file ), e );
333         }

383     private void computeCommand( ) throws MojoExecutionException {
384 
385         if ( protocExecutable == null )
386         {

388             return;
389         }
390

394                     protocExecutable.getGroupId( ),
395                     protocExecutable.getArtifactId( ),
396                     protocExecutable.getVersion( ),
397                     protocExecutable.getType( ),

400         getLog( ).info( String.format( "Using protoc from artifact %s.", protocArtifact ) );

402         final String protocDir = PathUtils.joinPaths( "target", "protobuf", "protoc" );
403         extractArtifact( protocArtifact, toAbsolutePath( protocDir ) );
404 
405         commandline.setExecutable( PathUtils.joinPaths( protocDir, "protoc" ) );
406         getLog( ).debug( String.format( "Set command to '%s'.", commandline.getExecutable( ) ) );

410         if ( Os.isFamily( Os.FAMILY_WINDOWS ) )

417     private String findExecutable( final File directory, final String basename ) throws MojoExecutionException {
418         final Collection<String> extensions = executableExtentionsByOs( );
419 
420         for ( final String ext : extensions )
421         {
422             final String execFilename = basename + ext;
423             final File potentialExecutable = new File( directory, execFilename );
424 
425             if ( potentialExecutable.isFile( ) )
426                 return potentialExecutable.getPath( );
427         }
428         throw new MojoExecutionException(
429             String.format( "Could not find executable %s in dir %s", basename, directory ) );
430     }

432     private Collection<String> executableExtentionsByOs( ) {
433         if ( Os.isFamily( Os.FAMILY_WINDOWS ) )
434             return WINDOWS_EXE;
435         if ( Os.isFamily( Os.FAMILY_UNIX ) )
436             return LINUX_EXE;
437         return Collections.<String>emptyList( );

459     private void addPluginToCommandLine( final ProtocPlugin protocPlugin ) throws MojoExecutionException {
460         // TODO Executable
461 
462         if ( protocPlugin.getExecutable( ) != null )
463         {
464             final Dependency dependency = protocPlugin.getDependency( );
465             final File executableFile;
466             if ( dependency != null )
467             {
468                 final Artifact artifact = resolveDependency( dependency );
469                 final String pluginDir = PathUtils.joinPaths(
470                     project.getBasedir( ).getPath( ),
471                     "target", "protobuf", "plugin", dependency.getArtifactId( ) );
472 
473                 final File extractPath = new File( pluginDir );
474                 final File artifactFile = artifact.getFile( );
475                 if ( artifactFile.isDirectory( ) )
476                 {
477                     getLog( ).info( "Artifact file is a directory, attempting to locate plugin archive in project." );
478                     final StringBuilder archiveNameBuilder = new StringBuilder( );
479 
480                     archiveNameBuilder
481                         .append( artifact.getArtifactId( ) )
482                         .append( "-" )
483                         .append( artifact.getVersion( ) );
484                     if ( !StringUtils.isEmpty( artifact.getClassifier( ) ) )
485                         archiveNameBuilder
486                             .append( "-" )
487                             .append( artifact.getClassifier( ) );
488                     archiveNameBuilder.append( "." )
489                         .append( artifact.getType( ) );
490 
491                     final File pluginArchive = new File( PathUtils.joinPaths(
492                         artifactFile.getAbsolutePath( ), "..", archiveNameBuilder.toString( ) ) );
493                     if ( pluginArchive.exists( ) )
494                     {
495                         getLog( ).info( "Found " + artifactFile.getAbsolutePath( ) );
496                         extractFile( pluginArchive, artifact.getType( ), extractPath );
497                     }
498                     else
499                     {
500                         throw new MojoExecutionException(
501                             String.format( "Cannot resolve plugin '%s' from directory %s.", artifact, artifactFile ) );
502                     }
503                 }
504                 else
505                 {
506                     extractArtifact( artifact, extractPath );
507                 }
508                 executableFile = new File( PathUtils.joinPaths(
509                     pluginDir, protocPlugin.getExecutable( ) ) );
510             }
511             else
512             {
513                 executableFile = new File(
514                     joinPaths(
515                         project.getBasedir( ).getPath( ),
516                         fixPath( protocPlugin.getExecutable( ) ) ) );
517             }
518             final String actualExecutable = findExecutable( executableFile.getParentFile( ), executableFile.getName( ) );
519 
520             addArgument( String.format( "--plugin=protoc-gen-%s=%s", protocPlugin.getPlugin( ), actualExecutable ) );
521         }
522 
523         addArgument( String.format( "--%s_out=%s",
524             protocPlugin.getPlugin( ),
525             protocPlugin.getOutputDirectory( ) ) );
526     }

581     private File archiveProtocol( final Iterable<String> directories, final String jarName )
582             throws MojoExecutionException {
583         final File protoArchiveFile = new File( project.getBasedir( ), PathUtils.joinPaths(
584             "target", "protobuf", jarName ) );
585         getLog( ).debug( String.format( "Archiving protocol sources to %s.", protoArchiveFile ) );
586 
587         Exception archiveException = null;
588         try
589         {
590             final Archiver archiver = archiverManager.getArchiver( "jar" );
591             archiver.setDestFile( protoArchiveFile );
592 
593             for ( final String sourceDirectory : directories )
594                 archiver.addDirectory( toAbsolutePath( sourceDirectory ) );
595 
596             archiver.createArchive( );
597             return protoArchiveFile;
598         }
599         catch ( final NoSuchArchiverException e )
600         {
601             archiveException = e;
602         }
603         catch ( final ArchiverException e )
604         {
605             archiveException = e;
606         }
607         catch ( final IOException e )
608         {
609             archiveException = e;
610         }
611 
612         getLog( ).error( archiveException.toString( ) );
613         throw new MojoExecutionException( PROTOCOL_ARCHIVE_ERROR, archiveException );
614     }
